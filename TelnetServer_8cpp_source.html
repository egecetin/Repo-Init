443 void TelnetSession::update()
444 {
445  ssize_t readBytes = 0;
446  std::array<char, DEFAULT_BUFLEN> recvbuf{};
447 
448  // Reset stats
449  stats.uploadBytes = 0;
450  stats.downloadBytes = 0;
451  stats.successCmdCtr = 0;
452  stats.failCmdCtr = 0;
453 
454  // Receive
455  readBytes = recv(m_socket, recvbuf.data(), DEFAULT_BUFLEN, 0);
456 
457  // Check for errors from the read
458  if (readBytes < 0 && errno != EAGAIN)
459  {
460  close(m_socket);
461  }
462  else if (readBytes > 0)
463  {
464  stats.downloadBytes += static_cast<size_t>(readBytes);
465 
466  // Update last seen
467  lastSeenTime = std::chrono::system_clock::now();
468 
469  // Echo it back to the sender
470  echoBack(recvbuf.data(), static_cast<size_t>(readBytes));
471 
472  // we've got to be careful here. Telnet client might send null characters for New Lines mid-data block. We need
473  // to swap these out. recv is not null terminated, so its cool
474  for (size_t i = 0; i < static_cast<size_t>(readBytes); i++)
475  {
476  if (recvbuf[i] == ASCII_NULL) // NOLINT(cppcoreguidelines-pro-bounds-constant-array-index)
477  {
478  // New Line constant
479  recvbuf[i] = ASCII_LF; // NOLINT(cppcoreguidelines-pro-bounds-constant-array-index)
480  }
481  }
482 
483  // Add it to the received buffer
484  m_buffer.append(recvbuf.data(), static_cast<unsigned int>(readBytes));
485  // Remove telnet negotiation sequences
486  stripNVT(m_buffer);
487 
488  bool requirePromptReprint = false;
489  if (m_telnetServer->interactivePrompt())
490  {
491  // Read up and down arrow keys and scroll through history
492  if (processCommandHistory(m_buffer))
493  {
494  requirePromptReprint = true;
495  }
496  stripEscapeCharacters(m_buffer);
497 
498  // Remove characters
499  if (processBackspace(m_buffer))
500  {
501  requirePromptReprint = true;
502  }
503  // Complete commands
504  if (processTab(m_buffer))
505  {
506  requirePromptReprint = true;
507  }
508  }
509 
510  // Process commands
511  auto lines = getCompleteLines(m_buffer);
512  for (const auto &line : lines)
513  {
514  if (m_telnetServer->newLineCallBack())
515  {
516  if (m_telnetServer->newLineCallBack()(shared_from_this(), line))
517  {
518  ++stats.successCmdCtr;
519  }
520  else
521  {
522  ++stats.failCmdCtr;
523  }
524  addToHistory(line);
525  }
526  }
527 
528  if (m_telnetServer->interactivePrompt() && requirePromptReprint)
529  {
530  eraseLine();
531  sendPromptAndBuffer();
532  }
533  }
534 }

679  fd_set readSet;
680  FD_ZERO(&readSet);
681  FD_SET(m_listenSocket, &readSet);
682  timeval timeout{};
683  timeout.tv_sec = 0; // Zero timeout (poll)
684  timeout.tv_usec = 0;
685

690  serverStats.processingTimeStart = std::chrono::high_resolution_clock::now();
691

693  if (select(m_listenSocket + 1, &readSet, nullptr, nullptr, &timeout) > 0)
694  {
695  if (acceptConnection())

697  newConnectionAccept = true;
698  }
699  else
700  {
701  newConnectionRefused = true;
702  }
703  }

706  for (size_t idx = 0; idx < m_sessions.size(); ++idx)
707  {
708  // Update session

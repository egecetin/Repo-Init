145void ZeroMQMonitor::stopMonitoring()
146{
147 _shouldStop.test_and_set();
148#ifdef ZMQ_EVENT_MONITOR_STOPPED
149 abort();
150#endif
151
152 // Join the thread
153 if (_monitorThread && _monitorThread->joinable())
154 {

157 }
158
159 spdlog::info("Monitor stopped");
